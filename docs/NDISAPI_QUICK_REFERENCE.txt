// Enable filtering on an adapter

    AdapterHandle: adapters.AdapterList[0].Handle,
    Flags: ndisapi.MSTCP_FLAG_SENT_TUNNEL | ndisapi.MSTCP_FLAG_RECV_TUNNEL,
}
api.SetAdapterMode(&mode)

// ... do packet interception work ...

// FLUSH PACKETS (your method!)
err := api.FlushAdapterPacketQueue(adapters.AdapterList[0].Handle)
if err != nil {
    log.Error("Failed to flush queue", err)
}

// Disable filtering (reset)
mode.Flags = 0
api.SetAdapterMode(&mode)

api.Close()
